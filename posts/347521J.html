def get_next(t):
    next_list = [-1] * len(t)
    j = -1
    i = 0
    s = t
    while i < len(s):
        if j == -1 or s[i] == t[j]:
            i += 1
            j += 1
            next_list[i] = j
        else:
            j = next_list[j]
    return next_list
